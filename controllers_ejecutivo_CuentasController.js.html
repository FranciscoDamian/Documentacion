/**
 * Controller for create cards table.
 * @class CuentasController
 * */
const { errorHandler } = require('../../helpers/errorHelper')
const { ok, internalError } = require( '../../helpers/statusCodes' )
const { _GET1, _POST, _PUT, _GET } = require( '../../helpers/errors' )
const {Tarjetas}  = require('../../models')
const {Op} = require('sequelize')

/** 
 * Get all card list for a client.
 * @function getAll
 * @memberOf CuentasController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {string} req.query.cuenta - Client number.
 * @return {Response} Response.
 * */
exports.getAll = async (req, res, next) => {
    try 
    {  
        const {cuenta } = req.query
        const where = {}
        if (cuenta) where.notarjeta = {[Op.iLike]:`%${cuenta}%`}
        const success = await Tarjetas.findAll({
            where: {
                ...where
            }
        }) 
        // console.log(success)
        return res.status( ok ).json(
        {
            message: _GET.tarjetas.success.ES,
            status: ok,
            success
        })
    }
    catch ( error ){
        next(error)
        // return res.status( internalError ).json( errorHandler ( error ) )
    }
}

/** 
 * Get one card list for a client.
 * @function getOne
 * @memberOf CuentasController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {string} req.body.notarjeta - Card number.
 * @return {Response} Response.
 * */
exports.getOne = async (req, res) => {
    try 
    {  
        const { notarjeta } = req.body
        const objTarjeta = await Tarjetas.findOne({
            where: { notarjeta }
        })
        return res.status( ok ).json( 
            { 
                message: _GET1.cuentas.success.ES,
                status: ok,
                success: objTarjeta
            })
    }
    catch ( error ){
        return res.status( internalError ).json( errorHandler ( error ) )
    }
}

/** 
 * Create a debit card for a client.
 * @function createDebit
 * @memberOf CuentasController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {Number} req.body.ClienteId - Client id.
 * @return {Response} Response.
 * */
exports.createDebit = async ( req, res ) => {
    try 
    {  
        const {ClienteId} = req.body
        const objTarjeta = await Tarjetas.create({
            tipo: 'Debito',
            notarjeta: Math.round(Math.random()* 1000000000000),
            comision: '5%',
            fechavencimiento: '2030-12-31',
            saldo: 0,
            limite: false,
            estado: true,
            ClienteId
        }) 
        return res.status( ok ).json( 
            { 
                message: _POST.cuentas.success.ES,
                status: ok,
                success: objTarjeta
            })
    }
    catch ( error ){
        return res.status( internalError ).json( errorHandler ( error ) )
    }
}

/** 
 * Create a new debit cart from one older.
 * @function updateOne
 * @memberOf CuentasController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {Number} req.body.id - Card id.
 * @param {Number} req.body.estado - New state of card.
 * @param {Number} req.body.fechavencimiento - New date of card.
 * @return {Response} Response.
 * */
exports.updateOne = async ( req, res ) => {
    try 
    {  
        const { id, fechavencimiento, estado} = req.body
        const objTarjeta = await Tarjetas.update({
            notarjeta: Math.round(Math.random()* 1000000000000),
            fechavencimiento,
            estado
        }, {
            where: { id }
        })
        return res.status( ok ).json( 
            { 
                message: _PUT.cuentas.success.ES,
                status: ok,
                success: objTarjeta
            })
    }
    catch ( error ){
        return res.status( internalError ).json( errorHandler ( error ) )
    }
}
