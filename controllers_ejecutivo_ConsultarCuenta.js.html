/**
 * Controller for request accounts.
 * @class ConsultarCuentaController
 * */
const { errorHandler } = require("../../helpers/errorHelper");
const { ok, internalError } = require("../../helpers/statusCodes");
const { _GET1, _POST, _PUT } = require("../../helpers/errors");
const { Op } = require("sequelize");

const { sequelize } = require("../../models");
const EstadoCuenta = require("../../models").Tarjetas;
const Cliente = require("../../models").Clientes;
const Transacciones = require("../../models").Transacciones;

/** 
 * Get all accounts from EstadoCuenta.
 * @function getAllCuentas
 * @memberOf ConsultarCuentaController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {Number} req.body.curp - Client CURP.
 * @param {Number} req.body.notarjeta - Client card number.
 * @return {Response} Response.
 * */
    exports.getAllCuentas = async (req, res) => {
    try {
        const { curp, notarjeta } = req.body;
        const cliente = await Cliente.findOne({
        attributes: ["id"],
        where: {
            curp,
        },
        });
        const consulta = cliente.dataValues.id;
        const cuentas = await EstadoCuenta.findAll({
        attributes: ["tipo", "notarjeta", "fechavencimiento", "estado"],
        where: {
            ClienteId: consulta,
        },
        });
        return res.status(ok).json({
        message: _GET1.cuentas.success.ES,
        status: ok,
        cuentas,
        });
    } catch (error) {
        return res.status(internalError).json({
        message: _GET1.cuentas.error.ES,
        status: internalError,
        });
    }
    };


    /** 
 * Get all the records of the period.
 * @function getPeriodo
 * @memberOf ConsultarCuentaController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {String} req.body.notarjeta - Card number.
 * @param {Date} req.body.FechaInincial - First date range number.
 * @param {Date} req.body.FechaFinal - Last date range number.
 * @return {Response} Response.
 * */
    exports.getPeriodo = async (req, res) => {
    try { 
        const { FechaInicial, FechaFinal, notarjeta } = req.body; 
        const estadocuenta = await EstadoCuenta.findOne({
        include: [
            {
            model: Cliente,
            attributes: [
                "nombre",
                "apellido",
                "rfc",
                "curp",
                "domicilio",
                "noexterior",
                "cp",
                "colonia",
                "municipio",
                "ciudad",
                "estado",
            ],
            where: {
                id: req.body.ClienteId,
            },
            },
        ],
        attributes: ["id", "notarjeta", "tipo", "saldo", "estado", "ClienteId"],
        where: { 
            notarjeta,
        },
        }); 
        const transacciones = await Transacciones.findAll({
        attributes: ["tipo", "fechatransaccion", "cantidad"],
        where: {
            TarjetaId: estadocuenta.dataValues.id,
            ClienteId: estadocuenta.dataValues.ClienteId,
            fechatransaccion: {
            [Op.between]: [FechaInicial, FechaFinal],
            },
        },
        }); 
        return res.status(ok).json({
        message: _GET1.periodo.success.ES,
        status: ok,
        estadocuenta,
        transacciones,
        });
    } catch (error) {
        return res.status(internalError).json({
        message: _GET1.periodo.error.ES,
        status: internalError,
        });
    }
    };
