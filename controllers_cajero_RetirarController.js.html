/**
 * @class RetirarController

const { ok, notFound, internalError } = require( '../../helpers/statusCodes' )
const { _PUT } = require( '../../helpers/errors' )
const { errorHandler, error400Handler } = require( '../../helpers/errorHelper' )

const Tarjetas  = require( '../../models' ).Tarjetas
const Transacciones = require('../../models').Transacciones

/** 
 * Generate a withdraw from a card.
 * @function updateOne
 * @memberOf RetirarController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @param {String} req.body.notarjeta - Card number.
 * @param {Number} req.body.saldo - Amount of money to withdraw.
 * @return {Response} res - Response.
 * */
exports.updateOne = async ( req, res ) => {
    const tipo = 'Retiro'
    const { notarjeta, saldo } = req.body
    let cantidad = parseFloat( saldo ).toFixed( 2 )
    let fechatransaccion = new Date();

    try {
        const consulta = await Tarjetas.findOne(
            {
                attributes: ['id', 'saldo', 'ClienteId'],
                where: { notarjeta }
            },
        )
    
    console.log( consulta.dataValues.saldo)
    console.log( cantidad )
    console.log( cantidad > consulta.dataValues.saldo )

    if ( consulta == null )  return res.status( notFound ).json( error400Handler( { x:0 } ) )
    if ( cantidad > consulta.dataValues.saldo ) {
        return res.status( ok ).json( {
                message: _PUT.retirar.error.ES,
                status: notFound,
                success: false
            }
        )
    }
    
    let saldoNuevo =  consulta.dataValues.saldo - parseFloat( cantidad )
    let idcliente  =  consulta.dataValues.ClienteId
    let idtarjeta = consulta.dataValues.id
    
    const success = await Tarjetas.update({ saldo: saldoNuevo }, { where: { notarjeta } } )

    if ( success == 0 ) return res.status( notFound ).json( error400Handler( { x:0 } ) ) 

        const objTransaccion = {
            tipo,
            fechatransaccion : fechatransaccion.toISOString(),
            cantidad,
            ClienteId: idcliente,
            TarjetaId: idtarjeta
        }

        console.table(objTransaccion)
        

        logTransaccion = await Transacciones.create( objTransaccion )
        console.log(logTransaccion) 

    return res.status( ok ).json(
        {
            message: _PUT.depositar.success.ES,
            status: ok,
            success
        }
    )
    } catch ( error ) { return res.status( internalError ).json( errorHandler( error ) ) }
}
