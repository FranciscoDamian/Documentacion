const express = require( 'express' );

/**
 * All the routes that work in the project on Bankmex need a prefix "/api/v1/", and all need to be auth to access, except for login route.
 * @module api
 */
const router = express.Router();

const api = require( '../helpers/api' )

// middleware auth
const auth = require( '../middlewares/auth.middleware' );

// Controllers
const depositarController = require( '../controllers/cajero/DepositarController' ); 
const retirarController = require( '../controllers/cajero/RetirarController' );
const reponerTarjetaController = require( '../controllers/cajero/ReponerTarjetaController' );
const dineroDisponible = require( '../controllers/cajero/DineroDisponible' );
const dataCliente = require( '../controllers/cajero/DataController' );
const avalesController = require('../controllers/ejecutivo/AvalesController'); 
const tarjetasController = require('../controllers/ejecutivo/CuentasController');

/**
 * Deposit money from a card
 * @name Transacciones card deposit 
 * @function login
 * @memberOf api/v1
 * @path {PUT} /api/v1/depositar
 * @auth This route requires HTTP Bearer Authentication.
 * @body {String} notarjeta - Card number.
 * @body {Number} saldo - Amount to deposit.
 * @service EmpleadoController
 */
router.put( `${ api.path }${ api.cajero.depositar }`, auth.cajero, depositarController.updateOne )

/**
 * Withdraw money from a card
 * @name Transacciones card withdraw
 * @path {PUT} /api/v1/retirar
 * @auth This route requires HTTP Bearer Authentication.
 * @body {String} notarjeta - Card number.
 * @body {Number} saldo - Amount to withdraw.
 */
router.put( `${ api.path }${ api.cajero.retirar }`, auth.cajero, retirarController.updateOne )

/**
 * A reposition for a card 
 * @name Tarjeta reposition
 * @path {PUT} /api/v1/retirar
 * @auth This route requires HTTP Bearer Authentication.
 * @body {String} notarjeta - Card number.
 * @body {String} motivo - Reason of reposition.
 */
router.put( `${ api.path }${ api.cajero.reponertarjera }`, auth.cajero, reponerTarjetaController.updateOne )

/**
 * Available money for an card
 * @name Tarjeta available money
 * @path {GET} /api/v1/login
 * @auth This route requires HTTP Bearer Authentication.
 */
router.get( `${ api.path }${ api.cajero.dinerodisponible }`, auth.cajero, dineroDisponible.getAll )

/**
 * Login
 * @name Login 
 * @path {POST} /api/v1/login
 * @body {String} username - User's username or nickname.
 * @body {String} password - User's password.
 */
router.post( `${ api.path }${ api.login }`, empleadoController.login );

/**
 * Get all Empleados
 * @name Empleados get all
 * @path {GET} /api/v1/empleados
 * @auth This route requires HTTP Bearer Authentication.
 * @query {Number} [page=0] - Current page.
 * @query {Number} [amount=10] - quantity of records.
 * @query {String} [sort=null] - DESC or ASC or null.
 * @query {Number} [role=null] - Role id.
 * @query {String} [search=null] - fragment of employee's name.

/**
