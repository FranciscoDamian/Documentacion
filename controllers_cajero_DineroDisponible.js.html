/**
 * @class DineroDisponibleController
 */
const { ok, notFound, internalError } = require( '../../helpers/statusCodes' )
const { _GET } = require( '../../helpers/errors' )
const { errorHandler } = require( '../../helpers/errorHelper' )

const Denominaciones  = require( '../../models' ).Denominaciones

/** 
 * Get record of the amount of money in all accounts.
 * @function getAll
 * @memberOf DineroDisponibleController
 * @param {Request} req - Request.
 * @param {Response} res - Response.
 * @param {Middleware} next - Middleware next method.
 * @return {Response} res - Response.
 * */
exports.getAll = async ( _, res ) => {
    try 
    {
    
        const success = await Denominaciones.findAll() 
        console.log(success)
        return res.status( ok ).json(
        {
            message: _GET.denominaciones.success.ES,
            status: ok,
            success
        })

    } catch ( error ) { 
        console.log(error)
        return res.status( internalError ).json( errorHandler( error ) ) }
}
